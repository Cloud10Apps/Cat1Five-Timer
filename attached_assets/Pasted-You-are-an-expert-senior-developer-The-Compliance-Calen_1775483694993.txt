Replace it ENTIRELY with this new version that matches ANY of the three

    if (params.data.month && params.data.year) {
      const startDate = dayjs()
        .year(params.data.year)
        .month(params.data.month - 1)
        .startOf("month")
        .format("YYYY-MM-DD");
      const endDate = dayjs()
        .year(params.data.year)
        .month(params.data.month - 1)
        .endOf("month")
        .format("YYYY-MM-DD");
      conditions.push(
        or(
          and(
            gte(inspectionsTable.nextDueDate, startDate),
            lte(inspectionsTable.nextDueDate, endDate)
          ),
          and(
            gte(inspectionsTable.scheduledDate, startDate),
            lte(inspectionsTable.scheduledDate, endDate)

            lte(inspectionsTable.completionDate, endDate)

Note the ! after the closing ) of or() — this is needed because Drizzle's 
or() can return undefined when given no args, but here we always pass args 
so it is safe. This satisfies TypeScript.

  const getDayActivities = (date: dayjs.Dayjs): CalendarActivity[] => {
    if (!inspections) return [];
    const activities: CalendarActivity[] = [];
    for (const insp of inspections) {
      const isCompleted = insp.status === "COMPLETED" || !!insp.completionDate;

      // Only show "due" chip if the inspection has NOT been completed
      if (!isCompleted && insp.nextDueDate && dayjs(insp.nextDueDate).isSame(date, "day"))
        activities.push({ insp, activityType: "due" });

      // Only show "scheduled" chip if the inspection has NOT been completed
      if (!isCompleted && insp.scheduledDate && dayjs(insp.scheduledDate).isSame(date, "day"))
        activities.push({ insp, activityType: "scheduled" });

      // Always show "completed" chip on the completion date
      if (insp.completionDate && dayjs(insp.completionDate).isSame(date, "day"))
        activities.push({ insp, activityType: "completed" });
    }
    return activities;
  };
