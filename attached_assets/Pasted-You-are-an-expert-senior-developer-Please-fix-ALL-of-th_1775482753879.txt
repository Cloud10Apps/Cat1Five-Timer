### 2E. Add subscription enforcement middleware
FILE: artifacts/api-server/src/middleware/auth.ts (add to existing file)
FILE: artifacts/api-server/src/routes/index.ts (apply middleware)

PROBLEM: Users can sign up and use the full app forever without paying.

FIX STEP 1 — Add requireActiveSubscription to auth.ts:

  export async function requireActiveSubscription(
    req: Request, 
    res: Response, 
    next: NextFunction
  ) {
    // Skip enforcement if Stripe is not configured (dev/demo mode)
    if (!process.env.STRIPE_SECRET_KEY) return next();

    const orgId = req.user!.organizationId;

    try {
      const { db, organizationsTable } = await import("@workspace/db");
      const { eq } = await import("drizzle-orm");
      const [org] = await db
        .select()
        .from(organizationsTable)
        .where(eq(organizationsTable.id, orgId));

      if (!org?.stripeSubscriptionId) {
        // No subscription — check if within grace period (14 days from org creation)
        const createdAt = new Date(org?.createdAt ?? 0);
        const daysSinceCreation = (Date.now() - createdAt.getTime()) / 86_400_000;
        if (daysSinceCreation <= 14) return next(); // 14-day free trial grace period
        return res.status(402).json({ 
          error: "Active subscription required.",
          code: "SUBSCRIPTION_REQUIRED"
        });
      }

      const { stripeStorage } = await import("./stripeStorage.js");
      const sub = await stripeStorage.getSubscription(org.stripeSubscriptionId);
      if (sub && (sub.status === "active" || sub.status === "trialing")) {
        return next();
      }

      return res.status(402).json({ 
        error: "Your subscription is inactive. Please update your billing.",
        code: "SUBSCRIPTION_INACTIVE"
      });
    } catch {
      // On error, fail open (don't block users due to Stripe outages)
      return next();
    }
  }
